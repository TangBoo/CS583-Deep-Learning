class Worker:
    def __init__(self,x,y,q=1,lr=0.1,lam=None):
        self.x=x
        self.y=y
        self.s=x.shape[0]
        self.d=x.shape[1]
        self.w=np.zeros((d,1))
        self.lam=lam
        self.q=q
        self.p=np.zeros((d,1))
        self.grads=0
        self.lr=lr
    def adagrad(self,t):
        t+=1
        smooth=1e-9
        sigma=((1/t)*self.grads+smooth)**0.5
        self.lr=self.lr/sigma
        
    def shuffle(self):
        n=self.x.shape[0]
        rand_indices=np.random.permutation(n)
        self.x=self.x[rand_indices]
        self.y=self.y[rand_indices]
    def set_param(self,w):
        self.w=w
        self.p=w
    def loss(self):
        yx=np.multiply(self.y,self.x)
        yxw=np.dot(yx,self.w)
        vec1=np.exp(-yxw)
        vec2=np.log(1+vec1)
        vec3=np.sum(vec2)
        return vec3
    def stochastic_objective_gradient(self,xi,yi):
        n,d=xi.shape
        yx=yi*xi
        yxp=float(np.dot(yx,self.p))
        loss=np.log(1+np.exp(-yxp))
        reg=(self.lam/2)*np.sum(self.p*self.p)
        obj=loss+reg
        g_loss=-yx.T/(1+np.exp(yxp))
        g=g_loss+self.lam*self.p
        return obj,g
    def sgd(self):
        n,d=self.x.shape
        objval=0
        for t in range(self.q):
            rand_indices=np.random.permutation(n)
            x_rand=self.x[rand_indices,:]
            y_rand=self.y[rand_indices,:]
            for i in range(n):
                xi=x_rand[i,:]
                yi=float(y_rand[i,:])
                obj,g=self.stochastic_objective_gradient(xi,yi)
                if t<1:
                    objval+=obj
                self.p-=self.lr*g
            self.lr*=0.9
#             objval /= n
        return (self.w-self.p),objval/n
    def gradient(self):
        n, d = self.x.shape
        yx = np.multiply(self.y,self.x)
        yxp= np.dot(yx,self.p)
        vec1 = np.exp(yxp)
        vec2 = np.divide(yx,1 + vec1)
        vec3 = -np.mean(vec2, axis=0).reshape(d, 1)
        g = vec3 +self.lam*self.p
        self.grads+=np.sum(g)**2
        return g
    def grad_desccent(self):
        for t in range(self.q):
            g = self.gradient()
            self.p -=self.lr*g
        return self.p

class Server:
    def __init__(self,m,n,d,lr=0.1):
        self.m=m
        self.n=n
        self.d=d
        self.w=np.zeros((d,1))
        self.g=np.zeros((d,1))
        self.p=np.zeros((d,1))
        self.loss=0
        self.obj=0
        self.lr=lr
        self.grads=0
    def adagrad(self,t):
        t+=1
        smooth=1e-9
        sigma=((1/t)*self.grads+smooth)**0.5
        self.lr=self.lr/sigma
    def broadcast(self):
        return self.w
    def aggregate_avg(self,ps,losses):
        self.p=np.zeros((self.d,1))
        self.loss=0
        for k in range(self.m):
            self.p+=ps[k]
            self.loss+=losses[k]
        self.p=self.p/m
        self.grads+=np.sum(self.p)**2
    def objective(self,lam,is_sgd=False):
        reg=lam/2*np.sum(self.w*self.w)
        if is_sgd:
            self.obj=self.loss/self.m+reg
        else:
            self.obj=self.loss/self.n+reg
        return self.obj
    def agd(self):
        self.w=self.w-self.lr*self.p

import math
lam=1E-6
alpha=1E-1
max_epoch=50
def create_server_workers(m,x,y,p=1,lr=alpha,lam=None):
    n,d=x.shape
    s=math.floor(n/m)
    server=Server(m,n,d,lr=alpha)
    workers=[]
    for i in range(m):
        indices=list(range(i*s,(i+1)*s))
        worker=Worker(x[indices,:],y[indices,:],p,lr=alpha,lam=lam)
        workers.append(worker)
    return server,workers
